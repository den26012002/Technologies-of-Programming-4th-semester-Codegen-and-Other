memberCallExpression : (Identifier Dot)* (Identifier | typeId) (LeftParen expression? (Comma expression)* RightParen);

templateDeclaration : Template Less (typeId | ((Typename_ | Class) Identifier))? (Comma | typeId | ((Typename_ | Class) Identifier))* Greater;
